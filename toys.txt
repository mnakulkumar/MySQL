CREATE TABLE toys
(
toy_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
toy VARCHAR(20)
);

CREATE TABLE girls
(
girl_id INT NOT NULL AUTO_INCREMENT PRIMARY KEY,
girl VARCHAR(20),
toy_id INT NOT NULL,
CONSTRAINT toys_toy_id_fk
FOREIGN KEY(toy_id)
REFERENCES toys(toy_id)
);
